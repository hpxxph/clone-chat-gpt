/**
 * animation close modal window
 * @returns {void}
 */

function animatedCloseModal() {
  
  let result = modalContent.classList.add("close-modal");
     
    setTimeout(function(){
      modalContent.classList.remove("close-modal");
      modal.style.display = "none";
    }, 400);
}
